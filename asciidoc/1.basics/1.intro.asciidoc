Dockerfiles
===========


----
mkdir -p ldthw_1
cd ldthw_1
cat > Dockerfile << END
FROM ubuntu
RUN apt-get update -y
RUN apt-get install -y nginx
CMD bash
END
docker build -t firstimage .
----


----
docker run -ti firstimage
----
